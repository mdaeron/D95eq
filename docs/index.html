D95thermo

Estimate carbonate formation temperatures from dual clumped isotope measurements

1. Installation

1.1 Recommended method

The recommended way is to use via uv (https://docs.astral.sh/uv).

If you only want to run the command-line interface (CLI): after installing uv, this should be as simple as uvx D95thermo or uv tool install D95thermo.

If you want to import D95thermo in some Python code, once you are within a uv project (uv init), you can install the module with uv add D95thermo.

After installation, open a new shell window and try D95thermo --help.

1.2 Other methods

You can of course install globally via pip (pip install D95thermo), or only install the CLI using pipx (pipx install D95thermo).

2. Command-line interface

D95thermo also provides a command-line interface (CLI).

  94			+ a3 / (273.15 + T)**3

 109# 		print(_cd.data_string(dict(affine_coefs = _cd.uarray([b0, b1]))))

 124	a2 *= b1
 125	a3 *= b1
 126	a4 *= b1

 131def D4x_calib_function(
 132	T: (float | _uc.UFloat | _cd.uarray | ArrayLike),
 133	coefs: _cd.uarray,
 134	return_without_uncertainties: bool = False,
 135	ignore_calib_uncertainties: bool = False,
 136) -> (float | _uc.UFloat | _cd.uarray | ArrayLike):
 137	"""
 138	**Arguments**
 139	* `T`: temperature(s) for which to compute Δ<sub>4x</sub>
 140	* `return_without_uncertainties`: if `True`, returns D4x values without error propagation of any kind.
 141	* `ignore_calib_uncertainties`: whether to propagate calibration uncertainties.
 142	"""
 143	degs = _np.arange(coefs.size)
 144
 145	D4x = (
 146		_np.expand_dims(_cd.nv(coefs) if ignore_calib_uncertainties else coefs, 1) # shape = (coefs.size, 1)
 147		* _np.expand_dims((T+273.15)**-1, 0)                                       # shape = (1, T.size)
 148		** _np.expand_dims(degs, 1)                                                # shape = (coefs.size, 1)
 149	).sum(axis = 0 if isinstance(T, _np.ndarray) else None)
 150
 151	if D4x.ndim == 0:
 152		return D4x.tolist().n if return_without_uncertainties else D4x.tolist()
 153	return D4x.n if return_without_uncertainties else D4x

 156#### Plotting functions ####
 157def conf_ellipse(
 158	X: (_cd.uarray | _np.ndarray | _uc.UFloat | float),
 159	Y: (_cd.uarray | _np.ndarray | _uc.UFloat | float) = None,
 160	p: float = 0.95,
 161	CM: (_np.ndarray | None) = None,
 162	Xse: (_np.ndarray | float | None) = None,
 163	Yse: (_np.ndarray | float | None) = None,
 164	ax: (_ppl.Axes | None) = None,
 165	**kwargs,
 166) -> tuple:
 167	"""
 168	Plot the joint *p*-level confidence ellipses for the elements of (X, Y),
 169	and return a list of the `Ellipse` objects thus created.
 170
 171	**Arguments**
 172	* `X`: x values
 173	* `Y`: y values
 174	* `p`: confidence level
 175	* `CM`: covariance matrix of (X, Y); not needed if X and Y are of type
 176		[`uncertainties.UFloat`](https://pythonhosted.org/uncertainties/tech_guide.html).
 177		or if (`Xse`, `Yse`) are specified.
 178	* `Xse`, `Yse`: SE of X and Y; not needed if X and Y are of type
 179		[`uncertainties.UFloat`](https://pythonhosted.org/uncertainties/tech_guide.html)
 180		or if `CM` is specified.
 181	* `ax`: which instance of `matplotlib.axes.Axes` to draw in; use current axes if `ax` = `None`.
 182	* `kwargs`: passed to `matplotlib.patches.Ellipse()`
 183	"""
 184
 185
 186	r2 = _chi2.ppf(p, 2)
 187	kwargs = dict(fc = 'None', ec = 'k', lw = 0.7) | kwargs
 188
 189	if ax is None:
 190		ax = _ppl.gca()
 191
 192	out = []
 193
 194	for x, y in zip(
 195		*_cd.as_pair_of_uarrays(X, Y, CM = CM, Xse = Xse, Yse = Yse)
 196	):
 197		val, vec = _eigh(_uc.covariance_matrix((x, y)))
 198		width, height = 2 * (val[:, None] * r2)**0.5
 199		angle = _np.degrees(_np.arctan2(*vec[::-1, 0]))
 200
 201		out.append(
 202			ax.add_patch(
 203				_Ellipse(
 204					xy = (x.n, y.n),
 205					width = width.item(),
 206					height = height.item(),
 207					angle = angle,
 208					**kwargs,
 209				)
 210			)
 211		)
 212
 213	return (*out,)

 332		Tmin: float = 0.,

 334		NT: int = 101,
 335		Tmarkers: _np.typing.ArrayLike = [0, 25, 100, 250, 1000],
 336		kwargs_Tmarkers: dict = {},
 337		show_Tmarker_labels: bool = True,
 338		kwargs_Tmarker_labels: dict = {},
 339		show_Tmarker_ellipses: bool = False,
 340		kwargs_Tmarker_ellipses: dict = {},
 341		show_eqline: bool = True,

 365		* `show_Tmarker_ellipses`: whether to add confidence ellipses to T markers

 367		* `show_eqline`: whether to plot the equilibrium curve itself

 423			ec = 'k',

 438			NT
 439		)**-0.5 - 273.15

 517	def nearest_Teq(
 518		self,
 519		D47: _cd.uarray,
 520		D48: _cd.uarray,
 521		ignore_calib_uncertainties: bool = False,
 522		estimate_pdf: bool = False,
 523		N_qmc: int = 1000,
 524	):
 525		"""
 526		Returns a `correldata.uarray` of T values, each of which is the closest (in the OGLS sense)
 527		to one (Δ<sub>47</sub>, Δ<sub>48</sub>) pair considered independently of the others.
 528		Also returns an array of corresponding p-values taking into account errors in Δ<sub>47</sub>
 529		and Δ<sub>48</sub> (and any covariance between the two) as well as errors in the
 530		Δ<sub>47</sub> and Δ<sub>48</sub> calibrations.
 531
 532		This is both the fastest and the strongly recommended version of this calculation.
 533		It is expected to yield an `uarray` with reasonably accurate covariance between the
 534		`Teq` values, but also between `Teq` and all other variables. Usually, these `Teq`
 535		values are strongly correlated with the corresponding values of `D47`.
 536		"""
 537
 538		N = D47.size
 539		N47 = self.D47_coefs.size
 540		N48 = self.D48_coefs.size
 541		Teq = D47 * 0
 542
 543		_np.set_printoptions(threshold = _np.inf)
 544		_np.set_printoptions(linewidth = _np.inf)
 545
 546
 547		if estimate_pdf:
 548
 549			from scipy.stats import qmc
 550			from tqdm import trange
 551
 552			input_params = _cd.uarray([
 553				*D47,
 554				*D48,
 555				*self.D47_coefs,
 556				*self.D48_coefs,
 557			])
 558
 559			qmc_dist = qmc.MultivariateNormalQMC(
 560				mean = input_params.n*0,
 561				cov = input_params.cor,
 562			)
 563			qmc_draws = input_params.n + qmc_dist.random(N_qmc) * input_params.s
 564
 565			T_qmc = _cd.uarray(_np.zeros((N_qmc, N)))
 566
 567		for i in range(N):
 568			def fun(*args): # args = (D47, D48, *D47_calib_coefs, *D48_calib_coefs)
 569
 570				args = _np.array(args)
 571				D47_n = args[0]
 572				D48_n = args[1]
 573				D47_calib_coefs_n = args[-N48-N47:-N48]
 574				D48_calib_coefs_n = args[-N48:]
 575
 576				params = _lmfit.Parameters()
 577				a0, a2 = _D47_approx_calib_coefs
 578				params.add('Ti', value = ((D47_n - a0) / a2)**-.5 - 273.15)
 579
 580				D47_u = _cd.uarray([_uc.ufloat(D47_n, D47.s[i])])
 581				D48_u = _cd.uarray([_uc.ufloat(D48_n, D48.s[i])])
 582				D47_calib_coefs_u = _cd.uarray(_uc.correlated_values(D47_calib_coefs_n, self.D47_coefs.covar))
 583				D48_calib_coefs_u = _cd.uarray(_uc.correlated_values(D48_calib_coefs_n, self.D48_coefs.covar))
 584
 585				def cost_fun(
 586					p,
 587					ignore_calib_uncertainties = ignore_calib_uncertainties,
 588				):
 589					R = _cd.uarray(_np.concatenate((
 590						D47_u - D4x_calib_function(p['Ti'], D47_calib_coefs_u, ignore_calib_uncertainties = ignore_calib_uncertainties),
 591						D48_u - D4x_calib_function(p['Ti'], D48_calib_coefs_u, ignore_calib_uncertainties = ignore_calib_uncertainties),
 592					)))
 593
 594					invS = _np.linalg.inv(R.covar)
 595					L = _cholesky(invS)
 596
 597					return L @ R.n
 598
 599				minresult = _lmfit.minimize(cost_fun, params, method = 'least_squares', scale_covar = False, jac = '3-point')
 600				# slower but yields very similar results:
 601				# minresult = _lmfit.minimize(cost_fun, params, method = 'powell', scale_covar = False)
 602
 603				return minresult.params['Ti'].value
 604
 605			wrapped_fun = _uc.wrap(fun)
 606			Teq[i] = wrapped_fun(D47[i], D48[i], *self.D47_coefs, *self.D48_coefs)
 607
 608			if estimate_pdf:
 609				for k in trange(N_qmc):
 610					T_qmc[k,i] = fun(
 611						qmc_draws[k,i],
 612						qmc_draws[k,N+i],
 613						*qmc_draws[k,-N47-N48:-N48],
 614						*qmc_draws[k,-N48:],
 615					)
 616
 617		R = _cd.uarray(_np.concatenate((
 618			D47 - D4x_calib_function(Teq.n, self.D47_coefs, ignore_calib_uncertainties = ignore_calib_uncertainties),
 619			D48 - D4x_calib_function(Teq.n, self.D48_coefs, ignore_calib_uncertainties = ignore_calib_uncertainties),
 620		)))
 621
 622		p = _np.zeros((N,))
 623		for k in range(N):
 624			r = R[k::N]
 625			z2 = r.m
 626			p[k] = 1-_chi2.cdf(z2, 1)
 627
 628		if estimate_pdf:
 629			return Teq, p, T_qmc
 630		return Teq, p

 791		self,
 792		D47: _cd.uarray,
 793		D48: _cd.uarray,

 795		ignore_calib_uncertainties: bool = False,
 796		estimate_pdf: bool = False,
 797		N_qmc: int = 1000,
 798	):
 799
 800		D47 = _cd.uarray(D47)
 801		D48 = _cd.uarray(D48)
 802		N = D47.size
 803		N47c = self.D47_coefs.size
 804		N48c = self.D48_coefs.size
 805		T = D47 * 0
 806
 807		if estimate_pdf:
 808
 809			from scipy.stats import qmc
 810			from tqdm import trange
 811
 812			input_params = _cd.uarray([
 813				*D47,
 814				*D48,
 815				kinetic_slope,
 816				*self.D47_coefs,
 817				*self.D48_coefs,
 818			])
 819
 820			qmc_dist = qmc.MultivariateNormalQMC(
 821				mean = input_params.n*0,
 822				cov = input_params.cor,
 823			)
 824			qmc_draws = input_params.n + qmc_dist.random(N_qmc) * input_params.s
 825
 826			T_qmc = _cd.uarray(_np.zeros((N_qmc, N)))
 827
 828		for i in range(N):
 829
 830			# function to solve
 831			def fun(t, *args): # args = (D47, D48, kinetic_slope, *self.D47_coefs, *self.D48_coefs)
 832				return (
 833					args[1]
 834					- _np.sum([
 835						c / (t[0] + 273.15)**k
 836						for k,c in enumerate(args[-N48c:])
 837					])
 838					- args[2] * (
 839						args[0]
 840						- _np.sum([
 841							c / (t[0] + 273.15)**k
 842							for k,c in enumerate(args[-N47c-N48c:-N48c])
 843						])
 844					)
 845				)
 846
 847			def g(*args):
 848				return _fsolve(fun, [100.], args = args)[0]
 849
 850			wg = _uc.wrap(g)
 851
 852			T[i] = wg(
 853				D47[i],
 854				D48[i],
 855				kinetic_slope,
 856				*self.D47_coefs,
 857				*self.D48_coefs,
 858			)

 860			if estimate_pdf:
 861				for k in trange(N_qmc):
 862					T_qmc[k,i] = wg(
 863						qmc_draws[k,i],
 864						qmc_draws[k,N+i],
 865						qmc_draws[k,N*2],
 866						*qmc_draws[k,-N47c-N48c:-N48c],
 867						*qmc_draws[k,-N48c:],
 868					)
 869
 870		if estimate_pdf:
 871			return T, T_qmc
